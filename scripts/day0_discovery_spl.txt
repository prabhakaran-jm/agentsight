# AgentSight Day 0 audit discovery queries (run in Splunk Search after MCP activity)
# Query 1 — find any MCP-related sourcetypes
index=_* OR index=main OR index=_internal OR index=_audit
  (sourcetype=*mcp* OR source=*mcp* OR "mcp" OR "splunk_run_query" OR "Splunk_MCP" OR agentsight_day0)
  earliest=-15m
| head 50
| table _time index sourcetype source user username action info search

# Query 2 — mcp_server tool calls (primary)
index=_internal sourcetype=mcp_server rpc_method=tools/call tool_name=*
| spath
| table _time username tool_name status execution_time_seconds request_id source_ip message

# Query 3 — splunkd_access MCP HTTP
index=_internal sourcetype=splunkd_access "Splunk_MCP_Server"
| rex field=_raw "\"(?<http_method>\w+) (?<uri>[^ ]+) HTTP"
| table _time user http_method uri _raw

# Query 4 — audittrail SPL text
index=_audit sourcetype=audittrail action=search (info=completed OR info=granted)
| search search=*splunk_run_query* OR search=*agentsight_day0* OR search=*makeresults*
| table _time user action info search
